millnames = ['',' K',' M',' B']
def millify(n):
    if pd.notnull(n):
       n = float(n)
       millidx = max(0,min(len(millnames)-1,
                                int(math.floor(0 if n == 0 else math.log10(abs(n))/3))))
       return '{:.1f}{}'.format(n / 10**(3 * millidx), millnames[millidx])
    else:
       return n
